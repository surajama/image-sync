const fs = require('fs'); const https = require('https'); const WebSocket = require('ws'); // Read cert and key const server = https.createServer({ cert: fs.readFileSync('cert.pem'), key: fs.readFileSync('key.pem') }); const wss = new WebSocket.Server({ server }); let clients = new Set(); wss.on('connection', ws => { const id = Math.random().toString(36).substring(2, 8); ws._id = id; clients.add(ws); console.log(`🟢 Client connected: ${id} (Total: ${clients.size})`); ws.on('message', message => { console.log(`📨 From ${id}: ${message}`); for (const client of clients) { if (client !== ws && client.readyState === WebSocket.OPEN) { client.send(message); console.log(`➡️ Relayed from ${id} to ${client._id}`); } } }); ws.on('close', () => { clients.delete(ws); console.log(`🔴 Client disconnected: ${id} (Remaining: ${clients.size})`); }); }); // Start the server server.listen(3000, () => { console.log('🚀 Secure WebSocket server running port 3000'); });
